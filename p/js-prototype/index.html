Element.prototype.wrap = function (tag) {
  const isContains = /\s?id*=*".*?"|\s?class*=*".*?"/gi;
  const isContainID = /\s?id*=*".*?"/gi;
  const isContainClass = /\s?class*=*".*?"/gi;

  let $target = this;
  let $tag = tag
    .replace(/<\/.*?>/g, "")
    .replace(/>.*?</g, ">/<")
    .split("/");
  $tag.reverse();

  let $makeEl = [];

  $tag.forEach(function (element, idx) {
    let plainTag = element.replace(/\s?id*=*".*?"|\s?class*=*".*?"|[<,>]|\s/gi, "");
    $makeEl[idx] = document.createElement(plainTag);

    if (element.match(isContains)) {
      if (element.match(isContainID))
        $makeEl[idx].id = element
          .match(isContainID)
          .toString()
          .replace(/\s?id*=*"|"|\s/gi, "");

      if (element.match(isContainClass)) {
        let $plainClass = element
          .match(isContainClass)
          .toString()
          .replace(/\s?class*=*"|"/gi, "")
          .split(" ");

        $plainClass.forEach((insertClass) => {
          $makeEl[idx].classList.add(insertClass);
        });
      }
    }

    if (idx == 0) $makeEl[idx].innerHTML = $target.outerHTML;
    else $makeEl[idx].innerHTML = $makeEl[idx - 1].outerHTML;
  });

  $target.parentElement.insertBefore($makeEl[$tag.length - 1], $target);
  this.remove();
};

Element.prototype.unwrap = function () {
  let $target = this;
  let $parentEl = this.parentElement;

  let $insertTag = "";

  for (var i = 0; i < $parentEl.children.length; i++) {
    $insertTag += $parentEl.children[i].outerHTML;
  }

  $parentEl.insertAdjacentHTML("beforebegin", $insertTag);
  $parentEl.remove();
};
